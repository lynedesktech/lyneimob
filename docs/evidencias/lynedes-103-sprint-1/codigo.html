if (/(?:^|\n)\s*---\s*(?:\n|$)/.test(texto)) {
  const blocos = texto
    .split(/(?:^|\n)\s*---\s*(?:\n|$)/)
    .map((b) => b.trim())
    .filter((b) => b.length > 0)
  const resultado: string[] = []
  for (const bloco of blocos) {
    if (bloco.length <= maxCaracteres) resultado.push(bloco)

Saudacao baseada no horario + contexto temporalENTREGUE
src/lib/whatsapp/prompt-sdr.ts:14
Funcoes obterSaudacao() e contextoTemporal() injetam data/hora/saudacao no prompt usando timezone America/Sao_Paulo.
export function obterSaudacao(hora: number): string {
  if (hora >= 5 && hora < 12) return "Bom dia"
  if (hora >= 12 && hora < 18) return "Boa tarde"
  return "Boa noite"
}

export function contextoTemporal(): string {
  // Timezone fixo America/Sao_Paulo (UTC-3)
  const agora = new Date()
  const offsetSP = -3 * 60
  const localSP = new Date(agora.getTime() + (agora.getTimezoneOffset() + offsetSP) * 60000)
  const saudacao = obterSaudacao(localSP.getHours())
  return `Data atual: ${dia}/${mes}/${ano} | Hora: ${hora}:${minuto} | Saudacao: ${saudacao}`
}

Redis key absence_sent:{org}:{chat} com TTL de 20h evita repetir a mensagem quando lead manda varias msgs fora do horario. Mensagem configuravel por org via config_whatsapp.mensagem_fora_horario.
# Python (Railway)
async def ja_enviou_ausencia_hoje(org_id: str, chat_id: str) -> bool:
    r = await get_redis()
    key = f"absence_sent:{org_id}:{chat_id}"
    val = await r.get(key)
    return val is not None

async def marcar_ausencia_enviada(org_id: str, chat_id: str):
    r = await get_redis()
    key = f"absence_sent:{org_id}:{chat_id}"
    await r.set(key, "1", ex=20 * 60 * 60)  # TTL 20h
